#!/bin/bash

if [ '47' == '45' ]; then
  echo "Skipping deployment because target version (47) is already the current version"
  exit 0
fi

ID=$(aws deploy create-deployment \
    --application-name authdev2-authorizer-auth  \
    --deployment-group-name authDeploymentGroup  \
    --revision '{"revisionType": "AppSpecContent", "appSpecContent": {"content": "{\"Resources\":[{\"authdev2-api_gateway_authorizer\":{\"Properties\":{\"Alias\":\"authdev2-authorizer-alias-lambda-active\",\"CurrentVersion\":\"45\",\"Name\":\"authdev2-api_gateway_authorizer\",\"TargetVersion\":\"47\"},\"Type\":\"AWS::Lambda::Function\"}}],\"version\":\"0.0\"}", "sha256": "468819e312f5a944b403c4c11632de88c404eb3d5b702fae8ea9d211b91ba1e2"}}' \
    --output text \
    --query '[deploymentId]')


STATUS=$(aws deploy get-deployment \
    --deployment-id $ID \
    --output text \
    --query '[deploymentInfo.status]')

while [[ $STATUS == "Created" || $STATUS == "InProgress" || $STATUS == "Pending" || $STATUS == "Queued" || $STATUS == "Ready" ]]; do
    echo "Status: $STATUS..."
    STATUS=$(aws deploy get-deployment \
        --deployment-id $ID \
        --output text \
        --query '[deploymentInfo.status]')

    SLEEP_TIME=$((( $RANDOM % 5 ) + 60))

    echo "Sleeping for: $SLEEP_TIME Seconds"
    sleep $SLEEP_TIME
done

aws deploy get-deployment --deployment-id $ID

if [[ $STATUS == "Succeeded" ]]; then
    echo "Deployment succeeded."
else
    echo "Deployment failed!"
    exit 1
fi
