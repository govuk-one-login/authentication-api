#!/bin/bash

if [ '43' == '43' ]; then
  echo "Skipping deployment because target version (43) is already the current version"
  exit 0
fi

ID=$(aws deploy create-deployment \
    --application-name authdev2-send-otp-notification-auth  \
    --deployment-group-name authDeploymentGroup  \
    --revision '{"revisionType": "AppSpecContent", "appSpecContent": {"content": "{\"Resources\":[{\"authdev2-send-otp-notification-lambda\":{\"Properties\":{\"Alias\":\"authdev2-send-otp-notification-lambda-active\",\"CurrentVersion\":\"43\",\"Name\":\"authdev2-send-otp-notification-lambda\",\"TargetVersion\":\"43\"},\"Type\":\"AWS::Lambda::Function\"}}],\"version\":\"0.0\"}", "sha256": "607f1cb00519e5f870d0abb0b5cc806e12bdbbd37104c7c387c1bd49aa7879c3"}}' \
    --output text \
    --query '[deploymentId]')


STATUS=$(aws deploy get-deployment \
    --deployment-id $ID \
    --output text \
    --query '[deploymentInfo.status]')

while [[ $STATUS == "Created" || $STATUS == "InProgress" || $STATUS == "Pending" || $STATUS == "Queued" || $STATUS == "Ready" ]]; do
    echo "Status: $STATUS..."
    STATUS=$(aws deploy get-deployment \
        --deployment-id $ID \
        --output text \
        --query '[deploymentInfo.status]')

    SLEEP_TIME=$((( $RANDOM % 5 ) + 60))

    echo "Sleeping for: $SLEEP_TIME Seconds"
    sleep $SLEEP_TIME
done

aws deploy get-deployment --deployment-id $ID

if [[ $STATUS == "Succeeded" ]]; then
    echo "Deployment succeeded."
else
    echo "Deployment failed!"
    exit 1
fi
