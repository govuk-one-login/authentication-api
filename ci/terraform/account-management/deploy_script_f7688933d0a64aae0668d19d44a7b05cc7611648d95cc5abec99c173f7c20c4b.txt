#!/bin/bash

if [ '48' == '46' ]; then
  echo "Skipping deployment because target version (48) is already the current version"
  exit 0
fi

ID=$(aws deploy create-deployment \
    --application-name authdev2-email_sqs_lambda-auth  \
    --deployment-group-name authDeploymentGroup  \
    --revision '{"revisionType": "AppSpecContent", "appSpecContent": {"content": "{\"Resources\":[{\"authdev2-account-management-sqs-lambda\":{\"Properties\":{\"Alias\":\"authdev2-account-management-sqs-lambda-active\",\"CurrentVersion\":\"46\",\"Name\":\"authdev2-account-management-sqs-lambda\",\"TargetVersion\":\"48\"},\"Type\":\"AWS::Lambda::Function\"}}],\"version\":\"0.0\"}", "sha256": "f7688933d0a64aae0668d19d44a7b05cc7611648d95cc5abec99c173f7c20c4b"}}' \
    --output text \
    --query '[deploymentId]')


STATUS=$(aws deploy get-deployment \
    --deployment-id $ID \
    --output text \
    --query '[deploymentInfo.status]')

while [[ $STATUS == "Created" || $STATUS == "InProgress" || $STATUS == "Pending" || $STATUS == "Queued" || $STATUS == "Ready" ]]; do
    echo "Status: $STATUS..."
    STATUS=$(aws deploy get-deployment \
        --deployment-id $ID \
        --output text \
        --query '[deploymentInfo.status]')

    SLEEP_TIME=$((( $RANDOM % 5 ) + 60))

    echo "Sleeping for: $SLEEP_TIME Seconds"
    sleep $SLEEP_TIME
done

aws deploy get-deployment --deployment-id $ID

if [[ $STATUS == "Succeeded" ]]; then
    echo "Deployment succeeded."
else
    echo "Deployment failed!"
    exit 1
fi
